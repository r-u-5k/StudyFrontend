arrow function

    
        function func1(value) {
            return value;
        }
        console.log(func1('func1입니다.'));
        const func2 = (value) => {
            return value;
        };
        console.log(func2('func2입니다.'));
        /* 화살표 함수 생략 표기법 */
        
        //인수가 1개일 때 소괄호 생략 가능
        const func3 = (value) => {
            return value;
        };
        console.log(func3('func3입니다.'));
        //인수가 2개 이상일 때 소괄호 생략 불가능
        const func4 = (value1, value2) => {
            return value1 + value2;
        };
        console.log('func4(1,2)==> ' + func4(1, 3));
        //return문 생략
        //1행으로 반환하므로 {},return생략
        const func51 = (num1, num2) => num1 + num2;
        console.log('func51(10,20)==> ' + func51(10, 20));
        const func52 = (num1, num2) => {
            num1 + num2;
        };
        console.log('func52(10,20)==> ' + func52(10, 20));
        //객체반환
        const func61 = (val1, val2) => {
            const obj = {
                name: val1,
                age: val2,
            };
            return obj;
        };
        console.log('func61==> ', func61('김경호', 30));
        const func62 = (val1, val2) => {
            return {
                name: val1,
                age: val2,
            };
        };
        console.log('func62==> ', func62('김경호', 30));
        //()를 이용해 한행으로 모으기
        const func63 = (val1, val2) => ({
            name: val1,
            age: val2,
        });
        console.log('func63==> ', func63('김경호', 30));

객체 분할 대입 또는 비구조화 할당 (Destructuring Assignment)
